// This is a simple program in Tureasy to print "Hello World" on the console

// Function declaration in Tureasy starts with '$' and ends with '%'
$main
string s = "Hello World\n";               // 'string' is a datatype in Tureasy which is used to store strings (group of characters)
write s;                                  // 'write' is a command to print a string on the console
%                                         // '%' marks the end of a function
